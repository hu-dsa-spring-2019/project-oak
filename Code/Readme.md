def bubble_sort(array):
    index = len(array) - 1
    while index >= 0:
        for j in range(index):
            if array[j][1] > array[j+1][1]:
                array[j], array[j+1] = array[j+1], array[j]
            if array[j][1] == array[j+1][1]:
                if array[j][2] > array[j+1][2]:
                    array[j], array[j+1] = array[j+1], array[j]          
        index -= 1
    return array
